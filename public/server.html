Listening to Broadcasts:
Team Chat Messages: 
// Listen for team chat messages
websocket.onmessage = function(event) {
    const message = JSON.parse(event.data);
    
    if (message.type === 'team_message') {
        console.log('Chat from:', message.data.playerName);
        console.log('Message:', message.data.message);
        console.log('Time:', new Date(message.data.timestamp * 1000));
        console.log('Server:', message.data.serverName);
    }
};
Smart Entity State Changes: 
// Listen for entity state changes
websocket.onmessage = function(event) {
    const message = JSON.parse(event.data);
    
    if (message.type === 'entity_changed') {
        console.log('Entity ID:', message.data.entityId);
        console.log('Entity Name:', message.data.entityName);
        console.log('Is Active:', message.data.isActive);
        console.log('Server:', message.data.serverName);
    }
};
Server Connection Events: 
// Listen for server connection events
websocket.onmessage = function(event) {
    const message = JSON.parse(event.data);
    
    if (message.type === 'server_connected') {
        console.log('Server connected:', message.data.serverName);
    } else if (message.type === 'server_disconnected') {
        console.log('Server disconnected:', message.data.serverName);
    }
};
Entity Paired Events: 
// Listen for new entity pairings
websocket.onmessage = function(event) {
    const message = JSON.parse(event.data);
    
    if (message.type === 'entity_paired') {
        console.log('New entity paired:', message.data.message);
    }
};
